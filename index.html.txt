    <h2>Calculadora de Apuestas</h2>
    <label>Cuota combinada:</label>
    <input type="number" id="cuotaCombinada" step="0.01">
    <label>Importe apostado:</label>
    <input type="number" id="importe" step="0.01">
    <label>Cuota cobertura:</label>
    <input type="number" id="cuotaCobertura" step="0.01">
    <button onclick="calcular()">Calcular</button>
    
    <div class="resultado" id="resultado"></div>

    <script>
        function calcular() {
            let cuotaCombinada = parseFloat(document.getElementById("cuotaCombinada").value);
            let importe = parseFloat(document.getElementById("importe").value);
            let cuotaCobertura = parseFloat(document.getElementById("cuotaCobertura").value);
            let comision = 0.065; // 6.5% de comisión

            if (cuotaCombinada > 0 && importe > 0 && cuotaCobertura > 0) {
                let gananciaNeta = (cuotaCombinada * importe) - importe;
                let apuestaCobertura = gananciaNeta / (cuotaCobertura - 1);
                let apuestaCoberturaFinal = apuestaCobertura / (1 - comision);
                let beneficioFinal = (importe * cuotaCombinada) - importe - apuestaCoberturaFinal;
                let cuotaImplicita = (beneficioFinal + importe) / importe;

                document.getElementById("resultado").innerHTML = `
                    <p>Apuesta en contra: <strong>${apuestaCoberturaFinal.toFixed(2)} €</strong></p>
                    <p>Beneficio final: <strong>${beneficioFinal.toFixed(2)} €</strong></p>
                    <p>Cuota implícita final: <strong>${cuotaImplicita.toFixed(2)}</strong></p>
                `;
            } else {
                document.getElementById("resultado").innerHTML = "<p>Introduce valores válidos.</p>";
            }
        }
    </script>
